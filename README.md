<script>
const advisors = [
    "Andrew Krueger",
    "Oliver Crane",
    "Richard Bell",
    "Tonya Murray",
    "Zach Abbott",
    "Connor Amesse",
    "Christian Haboosh",
    "Yvette Wiley",
    "Vince Olsen",
    "House"
];

const goal = 15;

function createRow(name) {
    return `
    <tr>
        <td>${name}</td>
        <td><input type="number" value="0" min="0" onchange="updateTotals()"></td>
        <td><input type="number" value="0" min="0" onchange="updateTotals()"></td>
        <td><input type="number" value="0" min="0" onchange="updateTotals()"></td>
        <td class="total">0</td>
        <td>${goal}</td>
        <td>
            <div class="progress-bar">
                <div class="progress-fill"></div>
            </div>
        </td>
    </tr>
    `;
}

function loadTable() {
    const body = document.getElementById("tableBody");
    advisors.forEach(a => body.insertAdjacentHTML("beforeend", createRow(a)));
}

function updateTotals() {
    const rows = document.querySelectorAll("#leaderboard tbody tr");
    let teamTotal = 0;

    rows.forEach(row => {
        const active = Number(row.children[1].querySelector("input").value);
        const sold = Number(row.children[2].querySelector("input").value);
        const delivered = Number(row.children[3].querySelector("input").value);

        const total = active + sold + delivered;
        teamTotal += total;

        row.querySelector(".total").innerText = total;

        let percent = Math.min((total / goal) * 100, 100);
        row.querySelector(".progress-fill").style.width = percent + "%";
    });

    let teamPercent = Math.round((teamTotal / (goal * advisors.length)) * 100);

    document.getElementById("teamTotals").innerText =
        `Team Total: ${teamTotal} / Goal ${goal * advisors.length} → ${teamPercent}%`;
}

loadTable();
</script>
